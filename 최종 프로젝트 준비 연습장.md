
```python
            m = Movie()

            m.overview = movie.get("overview")
            m.poster_path = movie.get("poster_path")
            m.save()

            y = movie.get("genre_ids")

                a = y[h]
                m.genre_ids.add(a)
```

```python
            user = user,
            article = article,
            parent_comment = parent_comment
```
